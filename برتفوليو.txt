    <script>
        // System Audio state variable
        let audioSystemEnabled = true;
        let audioCtx = null;

        // Initialize Audio Context on user interaction to abide by browser security policies
        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            }
        }

        // Web Audio Synthesizer sound generator
        function playRetroSound(type) {
            if (!audioSystemEnabled) return;
            initAudio();
            if (!audioCtx) return;

            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);

            const now = audioCtx.currentTime;

            if (type === 'beep') {
                osc.type = 'square';
                osc.frequency.setValueAtTime(150, now);
                osc.frequency.exponentialRampToValueAtTime(800, now + 0.1);
                gain.gain.setValueAtTime(0.05, now);
                gain.gain.linearRampToValueAtTime(0, now + 0.1);
                osc.start(now);
                osc.stop(now + 0.12);
            } else if (type === 'success') {
                // Happy 8-bit sound
                osc.type = 'square';
                osc.frequency.setValueAtTime(440, now);
                osc.frequency.setValueAtTime(554.37, now + 0.08);
                osc.frequency.setValueAtTime(659.25, now + 0.16);
                osc.frequency.setValueAtTime(880, now + 0.24);
                gain.gain.setValueAtTime(0.05, now);
                gain.gain.linearRampToValueAtTime(0.05, now + 0.3);
                gain.gain.linearRampToValueAtTime(0, now + 0.45);
                osc.start(now);
                osc.stop(now + 0.45);
            } else if (type === 'error') {
                // Flat Error chord
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(120, now);
                osc.frequency.setValueAtTime(110, now + 0.05);
                gain.gain.setValueAtTime(0.08, now);
                gain.gain.linearRampToValueAtTime(0, now + 0.25);
                osc.start(now);
                osc.stop(now + 0.25);

        // Real-time Clock Taskbar display
        function updateClock() {
            const now = new Date();
            let hours = now.getHours();
            let minutes = now.getMinutes();
            const ampm = hours >= 12 ? 'مساءً' : 'صباحاً';
            hours = hours % 12;
            hours = hours ? hours : 12; // the hour '0' should be '12'
            minutes = minutes < 10 ? '0'+minutes : minutes;
            const strTime = hours + ':' + minutes + ' ' + ampm;
            document.getElementById('taskbar-clock').innerText = strTime;
        }
        setInterval(updateClock, 1000);
        updateClock();

        // Main Gemini API integration function
        async function callGemini(systemPrompt, userQuery) {
            const apiKey = ""; // Left empty as required by Canvas instructions.
            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: userQuery }] }],
                systemInstruction: {
                    parts: [{ text: systemPrompt }]
                }
            };
            
            try {
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                
                if (!response.ok) {
                    throw new Error("فشل الاتصال بخوادم Gemini API");
                }
                
                const result = await response.json();
                return result.candidates?.[0]?.content?.parts?.[0]?.text || "لم نتمكن من الحصول على رد مفيد.";
            } catch (error) {
                console.error("Gemini API Error:", error);
                return null;
            }
        }

        // Chatbot Integration Functions
        async function sendChatMessage() {
            const inputField = document.getElementById('ai-chat-input');
            const messageText = inputField.value.trim();
            if (!messageText) return;

            // Clear input and append user message
            inputField.value = "";
            appendMessage("مستخدم", messageText, "user");
            playRetroSound('click');

            // Show loading bubble for Clippy
            const loaderId = appendMessage("📎 دبّوس", "يفكر ويستحضر ذكريات التسعينات السريعة...", "bot-loading");

            // Build system prompt for Clippy clone (Arabic Retro Vibe)
            const systemPrompt = "أنت 'دبّوس الذكي' (Clippy القديم)، خبير عمل حر ومستشار تسويق مستقل بروح التسعينات الدافئة. تحدث باللهجة العربية الودية المطعمة ببعض طرائف المبرمجين والـ Freelancers. أجب على أسئلة المستقلين الجدد بإرشادات عملية وواضحة جداً لحل مشاكلهم (تسعير، عملاء، تفاوض، تسويق). اجعل ردودك مباشرة وضمن فقرة أو اثنتين مع نصائح مفيدة جداً ونظمها بنقاط.";
            
            const aiResponse = await callGemini(systemPrompt, messageText);
            
            // Remove loading and append answer
            document.getElementById(loaderId).remove();
            
            if (aiResponse) {
                appendMessage("📎 دبّوس", aiResponse, "bot");
                playRetroSound('success');
            } else {
                appendMessage("📎 دبّوس", "عذراً يا صديقي، خطوط الهاتف مشغولة (يبدو أن أحداً يرفع سماعة الهاتف والإنترنت Dial-up منقطع)! أعد المحاولة لاحقاً.", "bot");
                playRetroSound('error');
            }
        }

        function appendMessage(sender, text, type) {
            const chatDisplay = document.getElementById('ai-chat-display');
            const messageDiv = document.createElement('div');
            const msgId = "msg-" + Math.random().toString(36).substr(2, 9);
            messageDiv.id = msgId;
            messageDiv.className = `p-2 rounded max-w-full ${type === 'user' ? 'bg-blue-50 text-right border-r-4 border-blue-900' : 'bg-gray-100 text-right border-r-4 border-yellow-600'}`;
            
            messageDiv.innerHTML = `<span class="font-bold ${type === 'user' ? 'text-blue-950' : 'text-yellow-950'}">${sender}:</span><p class="mt-1 leading-relaxed whitespace-pre-line">${text}</p>`;
            
            chatDisplay.appendChild(messageDiv);
            chatDisplay.scrollTop = chatDisplay.scrollHeight;
            return msgId;
        }

        // Simple Drag and Drop functionality for retro windows title bars
        const dragBars = document.querySelectorAll('.drag-bar');
        dragBars.forEach(bar => {
            const win = bar.parentElement;
            let isDragging = false;
            let startX, startY, initialX, initialY;

            bar.addEventListener('mousedown', dragStart);
            bar.addEventListener('touchstart', dragStart, { passive: true });

            function dragStart(e) {
                isDragging = true;
                // Focus current window
                windowZIndex += 1;
                win.style.zIndex = windowZIndex;
                
                // Active Window visually highlight styling
                document.querySelectorAll('.win95-outset').forEach(w => w.classList.remove('window-active'));
                win.classList.add('window-active');

                const clientX = e.type === 'touchstart' ? e.touches[0].clientX : e.clientX;
                const clientY = e.type === 'touchstart' ? e.touches[0].clientY : e.clientY;

                startX = clientX;
                startY = clientY;
                initialX = win.offsetLeft;
                initialY = win.offsetTop;

                document.addEventListener('mousemove', drag);
                document.addEventListener('mouseup', dragEnd);
                document.addEventListener('touchmove', drag, { passive: false });
                document.addEventListener('touchend', dragEnd);
            }

            function drag(e) {
                if (!isDragging) return;
                
                // Prevent scrolling on touch devices during drag
                if (e.type === 'touchmove') e.preventDefault();

                const clientX = e.type === 'touchmove' ? e.touches[0].clientX : e.clientX;
                const clientY = e.type === 'touchmove' ? e.touches[0].clientY : e.clientY;

                const dx = clientX - startX;
                const dy = clientY - startY;

                win.style.left = (initialX + dx) + 'px';
                win.style.top = (initialY + dy) + 'px';
                win.style.right = 'auto'; // Break alignment bounds
            }

            function dragEnd() {
                isDragging = false;
                document.removeEventListener('mousemove', drag);
                document.removeEventListener('mouseup', dragEnd);
                document.removeEventListener('touchmove', drag);
                document.removeEventListener('touchend', dragEnd);
            }
        });
